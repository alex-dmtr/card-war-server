CREATE TABLE CardTypes
(
	ID serial,
	Type varchar(255)
);

CREATE TABLE Cards
(
	ID serial,
	Type varchar(255),
	CardType varchar(255),
	CardGFX varchar(255),
	Name varchar(255),
	Mana int,
	Attack int,
	Health int,
	Quote varchar(255)
);

CREATE TABLE Actions
(
	ID serial,
	Type varchar(255)
);

CREATE TABLE Scenarios
(
	ID serial,
	Type varchar(255)
);

CREATE TABLE LoadingScreens
(
	Image varchar(255)
);

CREATE TABLE Quotes
(
	Quote varchar(255)
);

CREATE TABLE Classes
(
	ID serial,
	Type varchar(255),
	Name varchar(255),
	Hero varchar(255),
	Cardback varchar(255),
	Cardfront varchar(255)
)
